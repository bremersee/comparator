15  public class WellKnownTextConverterTest {
16  
17    /**
18     * Convert.
19     */
20    @Test
21    public void convert() {
22      WellKnownTextConverter converter = new WellKnownTextConverter();
23  
24      ComparatorField field0 = new ComparatorField("field0", true, true, true);
25      ComparatorField field1 = new ComparatorField("field1", true, false, true);
26      ComparatorField field2 = new ComparatorField("field2", true, true, false);
27      ComparatorField field3 = new ComparatorField("field3", true, false, false);
28      ComparatorField field4 = new ComparatorField("field4", false, false, true);
29      ComparatorField field5 = new ComparatorField("field5", false, false, false);
30      List<ComparatorField> expected = Arrays.asList(field0, field1, field2, field3, field4, field5);
31  
32      ComparatorFields fields = converter.convert(
33          "field0,true,true,true"
34              + "|field1,true,false,true"
35              + "|field2,true,true,false"
36              + "|field3,true,false,false"
37              + "|field4,false,false,true"
38              + "|field5,false,false,false");
39      Assert.assertNotNull(fields);
40      Assert.assertEquals(expected.size(), fields.getFields().size());
41      for (int i = 0; i < expected.size(); i++) {
42        Assert.assertEquals(expected.get(i), fields.getFields().get(i));
43      }
44    }
45  
46    /**
47     * Convert and expect empty.
48     */
49    @Test
50    public void convertAndExpectEmpty() {
51      WellKnownTextConverter converter = new WellKnownTextConverter(ValueComparator::new);
52      ComparatorFields fields = converter.convert(null);
53      Assert.assertNotNull(fields);
54      Assert.assertTrue(fields.getFields().isEmpty());
55    }
