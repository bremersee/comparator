17  package org.bremersee.comparator.spring.converter;
18  
19  import static org.assertj.core.api.Assertions.assertThat;
20  import static org.assertj.core.api.InstanceOfAssertFactories.list;
21  
22  import java.util.List;
23  import org.bremersee.comparator.model.SortOrder;
24  import org.bremersee.comparator.model.SortOrders;
25  import org.bremersee.comparator.model.SortOrdersTextProperties;
26  import org.junit.jupiter.api.Test;
27  
28  /**
29   * The sort orders converter test.
30   *
31   * @author Christian Bremer
32   */
33  class SortOrdersConverterTest {
34  
35    /**
36     * Convert.
37     */
38    @Test
39    void convert() {
40      SortOrdersConverter converter = new SortOrdersConverter();
41  
42      SortOrders actual = converter.convert(
43          "field0,asc,true,true"
44              + ";field1,asc,false,true"
45              + ";field2,asc,true,false"
46              + ";field3,asc,false,false"
47              + ";field4,desc,false,true"
48              + ";field5,desc,false,false");
49  
50      SortOrder sortOrder0 = new SortOrder("field0", true, true, true);
51      SortOrder sortOrder1 = new SortOrder("field1", true, false, true);
52      SortOrder sortOrder2 = new SortOrder("field2", true, true, false);
53      SortOrder sortOrder3 = new SortOrder("field3", true, false, false);
54      SortOrder sortOrder4 = new SortOrder("field4", false, false, true);
55      SortOrder sortOrder5 = new SortOrder("field5", false, false, false);
56      List<SortOrder> expected = List.of(
57          sortOrder0, sortOrder1, sortOrder2, sortOrder3, sortOrder4, sortOrder5);
58  
59      assertThat(actual)
60          .extracting(SortOrders::getSortOrders, list(SortOrder.class))
61          .containsExactlyElementsOf(expected);
62    }
63  
64    /**
65     * Convert with properties.
66     */
67    @Test
68    void convertWithProperties() {
69      SortOrdersConverter converter = new SortOrdersConverter(
70          SortOrdersTextProperties.builder()
71              .sortOrderArgsSeparator("-:-")
72              .sortOrderSeparator("&&")
73              .caseSensitiveValue("cs")
74              .caseInsensitiveValue("cis")
75              .nullIsFirstValue("nif")
76              .nullIsLastValue("nil")
77              .build());
78  
79      SortOrders actual = converter.convert(
80          "-:-asc-:-cis-:-nif"
81              + "&&field1"
82              + "&&field2-:-desc"
83              + "&&field3-:-desc-:-cs"
84              + "&&field4-:-desc-:-cs-:-nif"
85              + "&&-:-desc"
86      );
87  
88      SortOrder sortOrder0 = new SortOrder(null, true, true, true);
89      SortOrder sortOrder1 = new SortOrder("field1", true, true, false);
90      SortOrder sortOrder2 = new SortOrder("field2", false, true, false);
91      SortOrder sortOrder3 = new SortOrder("field3", false, false, false);
92      SortOrder sortOrder4 = new SortOrder("field4", false, false, true);
93      SortOrder sortOrder5 = new SortOrder(null, false, true, false);
94      List<SortOrder> expected = List.of(
95          sortOrder0, sortOrder1, sortOrder2, sortOrder3, sortOrder4, sortOrder5);
96  
97      assertThat(actual)
98          .extracting(SortOrders::getSortOrders, list(SortOrder.class))
99          .containsExactlyElementsOf(expected);
100   }

102 }
