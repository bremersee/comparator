44    /**
45     * Test to sort.
46     *
47     * @param softly the soft assertions
48     */
49    @Test
50    void toSort(SoftAssertions softly) {
51      SortOrder sortOrder0 = new SortOrder("f0", true, true, true);
52      SortOrder sortOrder1 = new SortOrder("f1", false, false, false);
53      List<SortOrder> sortOrders = List.of(sortOrder0, sortOrder1);
54  
55      Sort sort = SortMapper.toSort(sortOrders);
56  
57      softly.assertThat(sort)
58          .isNotNull();
59  
60      Sort.Order sortOrder = sort.getOrderFor("f0");
61      softly.assertThat(sortOrder)
62          .isNotNull()
63          .extracting(Sort.Order::isAscending, BOOLEAN)
64          .isTrue();
65      softly.assertThat(sortOrder)
66          .isNotNull()
67          .extracting(Sort.Order::isIgnoreCase, BOOLEAN)
68          .isTrue();
69      softly.assertThat(sortOrder)
70          .isNotNull()
71          .extracting(Sort.Order::getNullHandling)
72          .isEqualTo(NullHandling.NULLS_FIRST);
73  
74      sortOrder = sort.getOrderFor("f1");
75      softly.assertThat(sortOrder)
76          .isNotNull()
77          .extracting(Sort.Order::isAscending, BOOLEAN)
78          .isFalse();
79      softly.assertThat(sortOrder)
80          .isNotNull()
81          .extracting(Sort.Order::isIgnoreCase, BOOLEAN)
82          .isFalse();
83      softly.assertThat(sortOrder)
84          .isNotNull()
85          .extracting(Sort.Order::getNullHandling)
86          .isEqualTo(NullHandling.NULLS_LAST);
87    }

217   /**
218    * Apply defaults.
219    *
220    * @param softly the soft assertions
221    */
222   @Test
223   void applyDefaults(SoftAssertions softly) {
224     softly.assertThat(SortMapper.applyDefaults((Sort) null, true, true, false))
225         .as("Apply defaults on null")
226         .isEmpty();
227 
228     Sort expected = Sort.by(Sort.Order.by("something"));
229     softly.assertThat(SortMapper.applyDefaults(expected, null, null, null))
230         .as("Apply defaults with no changes.")
231         .isEqualTo(expected);
232 
233     Sort sort = Sort.by(
234         Sort.Order.by("a").with(Direction.ASC),
235         Sort.Order.by("b").with(Direction.ASC));
236     Sort actual = SortMapper.applyDefaults(sort, false, null, null, "a");
237     softly.assertThat(actual.getOrderFor("a"))
238         .isNotNull()
239         .extracting(Sort.Order::isAscending, BOOLEAN)
240         .as("Apply defaults (direction) of property 'a'")
241         .isFalse();
242     softly.assertThat(actual.getOrderFor("b"))
243         .as("Apply defaults (direction) of property 'b' and expect no changes")
244         .isEqualTo(sort.getOrderFor("b"));
245 
246     sort = Sort.by(
247         Sort.Order.by("a").ignoreCase(),
248         Sort.Order.by("b"));
249     actual = SortMapper.applyDefaults(sort, null, false, null, "a");
250     softly.assertThat(actual.getOrderFor("a"))
251         .isNotNull()
252         .extracting(Sort.Order::isIgnoreCase, BOOLEAN)
253         .as("Apply defaults (case-handling) of property 'a")
254         .isFalse();
255     softly.assertThat(actual.getOrderFor("b"))
256         .as("Apply defaults (case-handling) of property 'b' and expect no changes")
257         .isEqualTo(sort.getOrderFor("b"));
258 
259     sort = Sort.by(
260         Sort.Order.by("a").with(NullHandling.NULLS_FIRST),
261         Sort.Order.by("b").with(NullHandling.NULLS_LAST));
262     actual = SortMapper.applyDefaults(sort, null, null, false, "a");
263     softly.assertThat(actual.getOrderFor("a"))
264         .isNotNull()
265         .extracting(Sort.Order::getNullHandling)
266         .as("Apply defaults (null-handling) of property 'a'")
267         .isEqualTo(NullHandling.NULLS_LAST);
268     softly.assertThat(actual.getOrderFor("b"))
269         .as("Apply defaults (null-handling) of property 'b' and expect no changes")
270         .isEqualTo(sort.getOrderFor("b"));
271   }
