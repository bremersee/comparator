17  package org.bremersee.comparator;
18  
19  import java.lang.reflect.Field;
20  import java.lang.reflect.Method;
21  import java.util.Optional;
22  
23  /**
24   * The default value extractor supports field names and paths as described in {@link
25   * org.bremersee.comparator.model.ComparatorField}. A field name can be just the field name (e. g.
26   * {@code firstName}), a method name, that is not generated by {@link
27   * ValueExtractor#getPossibleMethodNames(String)} (e. g. {@code toString}) or a path of field names
28   * separated by dots (.), e. g. {@code person.firstName}.
29   *
30   * @author Christian Bremer
31   */
32  @SuppressWarnings("WeakerAccess")
33  public class DefaultValueExtractor implements ValueExtractor {
34  
35    private final boolean throwingException;
36  
37    /**
38     * Instantiates a new default value extractor that will throw {@link ValueExtractorException}, if
39     * the given field cannot be found.
40     */
41    public DefaultValueExtractor() {
42      this(true);
43    }
44  
45    /**
46     * Instantiates a new default value extractor.
47     *
48     * @param throwingException if {@code true} and the given field cannot be found, {@link
49     *                          ValueExtractorException} will be thrown; otherwise {@code null} will
50     *                          be returned
51     */
52    public DefaultValueExtractor(boolean throwingException) {
53      this.throwingException = throwingException;
54    }
55  
56    @Override
57    public Object findValue(Object obj, String fieldPath) {
58      final String fieldIdentifier = trimFieldPath(fieldPath);
59      if (obj == null || fieldIdentifier == null || fieldIdentifier.length() == 0) {
60        return obj;
61      }
62  
63      final int index = fieldIdentifier.indexOf('.');
64      final String fieldName = index < 0 ? fieldIdentifier : fieldIdentifier.substring(0, index);
65      if (fieldName.length() == 0) {
66        return findValue(obj, fieldIdentifier.substring(index + 1));
67      }
68  
69      Object value;
70      Optional<Field> field = findField(obj.getClass(), fieldName);
71      if (field.isPresent()) {
72        value = invoke(field.get(), obj);
73      } else {
74        Optional<Method> method = findMethod(obj.getClass(), fieldName);
75        if (method.isPresent()) {
76          value = invoke(method.get(), obj);
77        } else if (throwingException) {
78          throw new ValueExtractorException("Field [" + fieldName + "] was not found on object ["
79              + obj + "].");
80        } else {
81          value = null;
82        }
83      }
84  
85      return index < 0 ? value : findValue(value, fieldIdentifier.substring(index + 1));
86    }
87  
88    private static String trimFieldPath(String field) {
89      if (field == null) {
90        return null;
91      }
92      String tmp = field
93          .replace("..", ".")
94          .trim();
95      while (tmp.startsWith(".")) {
96        tmp = tmp.substring(1).trim();
97      }
98      while (tmp.endsWith(".")) {
99        tmp = tmp.substring(0, tmp.length() - 1).trim();
100     }
101     return tmp;
102   }
103 
104 }
